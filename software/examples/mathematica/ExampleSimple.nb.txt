Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="XYZ"(*Change XYZ to the UID of your Temperature IR Bricklet 2.0*)

(*Create IPConnection and device object*)
ipcon=NETNew["Tinkerforge.IPConnection"]
tir=NETNew["Tinkerforge.BrickletTemperatureIRV2",uid,ipcon]
ipcon@Connect[host,port]

(*Get current ambient temperature (unit is 째C/10)*)
Print["Ambient Temperature: "<>ToString[N[Quantity[tir@GetAmbientTemperature[],"째C/10"]]]]

(*Get current object temperature (unit is 째C/10)*)
Print["Object Temperature: "<>ToString[N[Quantity[tir@GetObjectTemperature[],"째C/10"]]]]

(*Clean up*)
ipcon@Disconnect[]
ReleaseNETObject[tir]
ReleaseNETObject[ipcon]
